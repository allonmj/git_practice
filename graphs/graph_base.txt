class Graph implements Drawable{
	private int width, height;
	private int xMin, xMax;
	private int yMin, yMax;

	public Graph(){
		
	}

	@Override
	public void draw(){
		drawCrazily();
	}
	
	private void drawCrazily(){
		for (int i = 0; i< height; i++){
			for (int j = 0; j < width; j++){
				paint(i,j);
			}
		}
	}
}
